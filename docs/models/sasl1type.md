# Sasl1Type

Authentication parameters to use when connecting to brokers. Using TLS is highly recommended.

## Example Usage

```typescript
import { Sasl1Type } from "cribl-control-plane/models";

let value: Sasl1Type = {
  password: "wOjTO00_gqDKYmW",
  textSecret: "<value>",
  clientSecret: "<value>",
  clientTextSecret: "<value>",
  certificateName: "<value>",
  certPath: "<value>",
  privKeyPath: "<value>",
  passphrase: "<value>",
  clientId: "<id>",
  tenantId: "<id>",
  scope: "<value>",
};
```
